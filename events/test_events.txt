country_event = {
	id = dreadnought.1
	immediate = {log = "[GetDateText]: [Root.GetName]: event dreadnought.1"}
	hidden = yes
	fire_only_once = yes
	trigger = {
		date > 1936.1.10
	}

	immediate = {
		GER = {
			news_event = {
				id = dreadnought.2
			}
		}
		every_country = {
			if = {
				limit = {
					NOT = {
						tag = GER
					}
				}
				news_event = {
					days = 2
					id = dreadnought.3
				}
			}
		}
	}
}
