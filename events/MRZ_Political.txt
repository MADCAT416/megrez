#Add popularity
country_event = {
	id = mrz_gen_politic.6
	immediate = {log = "[GetDateText]: [Root.GetName]: event mrz_gen_politic.6"}

	hidden = yes

immediate = {
	if = {
		limit = { ROOT = { has_country_flag = totalist_coalition } }
		ROOT = {
		set_variable = { totalist_popularity = ROOT.party_popularity@totalist }
		multiply_variable = { totalist_popularity = 100 }
		add_to_variable = { goverment_popularity = totalist_popularity }
